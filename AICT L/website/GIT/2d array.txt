#include<iostream>
using namespace std;
void SUM_2D(int A[][5],int row,int num);  //function prototype//
void SUM_2D(int A[][5],int row,int num){  //function definition//
	int sum[row];                          //sum is array take size of no of rows,a=no of rows// 
	int a;
	for(int i=0;i<row;i++){  //i= increment counter to procced the loop until the noof rows,sum[i]=0 for every value of i the block of array always start with 0// 
		sum[i]=0;          // to calculate each row sum  which is stored respective'i' block of sum//
		a=i;
		for(int j=0;j<5;j++){
			if(A[i][j]){
				if(num>A[i][j]){   //condition= a number less than the number of which you want sum then if statement will work if it is true.until the loop istrue//
			      sum[i]=sum[i]+A[i][j];
			}}}
		cout<<"Sum of row "<<++a<<":"<<sum[i]<<endl;  // sum of each row is displaying//	
		}
}
int main(){
	int row,num;
	cout<<"Enter the no of rows:"<<endl;
	cin>>row;
	int A[row][5];
	cout<<"Enter the elements of 2D array:"<<endl;
	for(int i=0;i<row;i++){
		for(int j=0;j<5;j++){   // entering the elements of array// 
			cin>>A[i][j];
		} 
		cout<<endl;
	}
	cout<<"Enter a number less than the number of which you want sum:"<<endl;
	cin>>num;
	SUM_2D(A,row,num); // function calling//
		
}
