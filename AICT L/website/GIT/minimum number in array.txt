#include<iostream>
using namespace std;
int mins(int A[10]);
int mins(int A[10]){
	int min;
	min=A[0];
	for(int i=0;i<10;i++){
		if(min>A[i]){
			min=A[i];
		}
	}
	return min;
}

int main(){
	int A[10];
	int count=0,min;
	cout<<"Enter ten values:"<<endl;
	for(int i=0;i<10;i++){
		cin>>A[i];
	}
	min=mins(A);
	for(int j=0;j<10;j++){
	
	if(A[j]>min){ 
	count++;
	}
}
cout<<"Minimum number:"<<min<<endl;
cout<<"Count greater than minimum:"<<count<<endl;
}
