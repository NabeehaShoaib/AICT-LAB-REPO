#include<iostream>
using namespace std;
float sum (float num1,float num2);
float sum (float num1,float num2)
{
	float c;
	c =num1+num2;
	return c;
}
float sub (float num1,float num2);
float sub (float num1,float num2)
{
	float c;
	c =num1-num2;
	return c;
}
float multiply (float num1,float num2);
float multiply (float num1,float num2)
{
	float c;
	c =num1*num2;
	return c;
}
float div (float num1,float num2);
float div (float num1,float num2)
{
	float c;
	c =num1/num2;
	return c;
}
int main()
{
